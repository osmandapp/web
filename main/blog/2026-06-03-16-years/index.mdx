export function getInitialBirthdayLanguage() {
  if (typeof window === 'undefined') {
    return 'en';
  }

  const params = new URLSearchParams(window.location.search);
  const langFromUrl = params.get('lang');

  if (langFromUrl && birthdayLanguages[langFromUrl]) {
    return langFromUrl;
  }

  const savedLang = window.localStorage.getItem('osmandBirthdayArticleLanguage');

  if (savedLang && birthdayLanguages[savedLang]) {
    return savedLang;
  }

  return 'en';
}

export function updateBirthdayLanguageUrl(lang) {
  if (typeof window === 'undefined') {
    return;
  }

  const url = new URL(window.location.href);
  url.searchParams.set('lang', lang);
  window.history.replaceState({}, '', url);
  window.localStorage.setItem('osmandBirthdayArticleLanguage', lang);
}

export function BirthdayLanguageSelect({lang, setLang}) {
  function handleChange(event) {
    const nextLang = event.target.value;
    setLang(nextLang);
    updateBirthdayLanguageUrl(nextLang);
  }

  return (
    <div className="birthday-language-select">
      <label htmlFor="birthday-language-select">Language</label>
      <select
        id="birthday-language-select"
        value={lang}
        onChange={handleChange}
        aria-label="Select article language"
      >
        {Object.entries(birthdayLanguages).map(([value, label]) => (
          <option key={value} value={value}>{label}</option>
        ))}
      </select>
    </div>
  );
}
